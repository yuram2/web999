3.2 for문을 이용해서 테이블을 출력하시오!

조건1. 1~25까지 출력하세요!

{
    let num = 1;
        let table = "<table>"; 

        for(let i=1; i<=5; i++ ){
            table += "<tr>";
                for( let j=1; j<=5; j++ ){
                    table += "<td>"+ num +"</td>";
                    num++;
                }
            table += "</tr>";
        }
        table += "</table>";
        document.write(table);
}
